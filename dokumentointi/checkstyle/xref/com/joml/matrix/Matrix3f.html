12  /**
13   * Matrix3f
14   * 
15   * Contains the definition of a 3x3 Matrix of floats, and associated functions to transform
16   * it. The matrix is column-major to match OpenGL's interpretation, and it looks like this:
17   * 
18   *      m00  m10  m20
19   *      m01  m11  m21
20   *      m02  m12  m22
21   * 
22   * @author Richard Greenlees
23   */

87      /** Multiplies this matrix by the supplied matrix. This matrix will be the left-sided one */
88      public void mul(Matrix3f right) {
89          set(this.m00 * right.m00 + this.m10 * right.m01 + this.m20 * right.m02,
90              this.m01 * right.m00 + this.m11 * right.m01 + this.m21 * right.m02,
91              this.m02 * right.m00 + this.m12 * right.m01 + this.m22 * right.m02,
92              this.m00 * right.m10 + this.m10 * right.m11 + this.m20 * right.m12,
93              this.m01 * right.m10 + this.m11 * right.m11 + this.m21 * right.m12,
94              this.m02 * right.m10 + this.m12 * right.m11 + this.m22 * right.m12,
95              this.m00 * right.m20 + this.m10 * right.m21 + this.m20 * right.m22,
96              this.m01 * right.m20 + this.m11 * right.m21 + this.m21 * right.m22,
97              this.m02 * right.m20 + this.m12 * right.m21 + this.m22 * right.m22);
98      }

100     /** Multiplies the left matrix by the right, and stores the results in dest. Does not modify the left or right matrices */
101     public static void mul(Matrix3f left, Matrix3f right, Matrix3f dest) {
102         dest.set(left.m00 * right.m00 + left.m10 * right.m01 + left.m20 * right.m02,
103                  left.m01 * right.m00 + left.m11 * right.m01 + left.m21 * right.m02,
104                  left.m02 * right.m00 + left.m12 * right.m01 + left.m22 * right.m02,
105                  left.m00 * right.m10 + left.m10 * right.m11 + left.m20 * right.m12,
106                  left.m01 * right.m10 + left.m11 * right.m11 + left.m21 * right.m12,
107                  left.m02 * right.m10 + left.m12 * right.m11 + left.m22 * right.m12,
108                  left.m00 * right.m20 + left.m10 * right.m21 + left.m20 * right.m22,
109                  left.m01 * right.m20 + left.m11 * right.m21 + left.m21 * right.m22,
110                  left.m02 * right.m20 + left.m12 * right.m21 + left.m22 * right.m22);
111     }

113     /** Multiplies the left matrix by the right, and stores the results in dest. Does not modify the left or right matrices. 
114     * <B>This is not alias safe so make sure dest is not the same object as the original or you WILL get incorrect results!</B> */
115     public static void mulFast(Matrix3f left, Matrix3f right, Matrix3f dest) {
116         dest.m00 = left.m00 * right.m00 + left.m10 * right.m01 + left.m20 * right.m02;
117         dest.m01 = left.m01 * right.m00 + left.m11 * right.m01 + left.m21 * right.m02;
118         dest.m02 = left.m02 * right.m00 + left.m12 * right.m01 + left.m22 * right.m02;
119         dest.m10 = left.m00 * right.m10 + left.m10 * right.m11 + left.m20 * right.m12;
120         dest.m11 = left.m01 * right.m10 + left.m11 * right.m11 + left.m21 * right.m12;
121         dest.m12 = left.m02 * right.m10 + left.m12 * right.m11 + left.m22 * right.m12;
122         dest.m20 = left.m00 * right.m20 + left.m10 * right.m21 + left.m20 * right.m22;
123         dest.m21 = left.m01 * right.m20 + left.m11 * right.m21 + left.m21 * right.m22;
124         dest.m22 = left.m02 * right.m20 + left.m12 * right.m21 + left.m22 * right.m22;
125     }

168     /** Returns the determinant of this Matrix */
169     public float determinant() {
170         return   ((m00 * m11 * m22)
171                + (m10 * m21 * m02)
172                + (m20 * m01 * m12))
173                - ((m20 * m11 * m02)
174                + (m00 * m21 * m12)
175                + (m10 * m01 * m22));
176     }

178     /** Inverts this matrix */
179     public void invert() {
180         float s = determinant();
181         
182         if (s == 0.0f) {
183             return;
184         }
185         s = 1.0f / s;
186 
187         set((m11 * m22) - (m21 * m12),
188             -((m01 * m22) - (m21 * m02)),
189             (m01 * m12) - (m11 * m02),
190             -((m10 * m22) - (m20 * m12)),
191             (m00 * m22) - (m20 * m02),
192             -((m00 * m12) - (m10 * m02)),
193             (m10 * m21) - (m20 * m11),
194             -((m00 * m21) - (m20 * m01)),
195             (m00 * m11) - (m10 * m01));    
196         mul(s);
197     }

199     /** Inverts the source matrix and stores the results in dest. Does not modify the source */
200     public static void invert(Matrix3f source, Matrix3f dest) {
201         float s = source.determinant();
202         if (s == 0.0f) {
203             return;
204         }
205         s = 1.0f / s;
206         dest.set(((source.m11 * source.m22) - (source.m21 * source.m12)) * s,
207                  -((source.m01 * source.m22) - (source.m21 * source.m02)) * s,
208                   ((source.m01 * source.m12) - (source.m11 * source.m02)) * s,
209                  -((source.m10 * source.m22) - (source.m20 * source.m12)) * s,
210                   ((source.m00 * source.m22) - (source.m20 * source.m02)) * s,
211                  -((source.m00 * source.m12) - (source.m10 * source.m02)) * s,
212                   ((source.m10 * source.m21) - (source.m20 * source.m11)) * s,
213                  -((source.m00 * source.m21) - (source.m20 * source.m01)) * s,
214                   ((source.m00 * source.m11) - (source.m10 * source.m01)) * s);
215     }

217     /** Inverts the source matrix and stores the results in dest. Does not modify the source
218     * <B>This is not alias safe so make sure dest is not the same object as the original or you WILL get incorrect results!</B> */
219     public static void invertFast(Matrix3f source, Matrix3f dest) {
220         float s = source.determinant();
221         if (s == 0.0f) {
222             return;
223         }
224         s = 1.0f / s;
225         
226         dest.m00 = ((source.m11 * source.m22) - (source.m21 * source.m12)) * s;
227         dest.m01 = -((source.m01 * source.m22) - (source.m21 * source.m02)) * s;
228         dest.m02 = ((source.m01 * source.m12) - (source.m11 * source.m02)) * s;
229         dest.m10 = -((source.m10 * source.m22) - (source.m20 * source.m12)) * s;
230         dest.m11 = ((source.m00 * source.m22) - (source.m20 * source.m02)) * s;
231         dest.m12 = -((source.m00 * source.m12) - (source.m10 * source.m02)) * s;
232         dest.m20 = ((source.m10 * source.m21) - (source.m20 * source.m11)) * s;
233         dest.m21 = -((source.m00 * source.m21) - (source.m20 * source.m01)) * s;
234         dest.m22 = ((source.m00 * source.m11) - (source.m10 * source.m01)) * s;
235     }

237     /** Inverts the source matrix and stores the results in dest. Does not modify the source
238     * <B>This is not alias safe so make sure dest is not the same object as the original or you WILL get incorrect results!</B> */
239     public static void invert(Matrix3f source, FloatBuffer dest) {
240         float s = source.determinant();
241         if (s == 0.0f) {
242             return;
243         }
244         s = 1.0f / s;
245         
246         dest.put(((source.m11 * source.m22) - (source.m21 * source.m12)) * s);
247         dest.put(-((source.m01 * source.m22) - (source.m21 * source.m02)) * s);
248         dest.put(((source.m01 * source.m12) - (source.m11 * source.m02)) * s);
249         dest.put(-((source.m10 * source.m22) - (source.m20 * source.m12)) * s);
250         dest.put(((source.m00 * source.m22) - (source.m20 * source.m02)) * s);
251         dest.put(-((source.m00 * source.m12) - (source.m10 * source.m02)) * s);
252         dest.put(((source.m10 * source.m21) - (source.m20 * source.m11)) * s);
253         dest.put(-((source.m00 * source.m21) - (source.m20 * source.m01)) * s);
254         dest.put(((source.m00 * source.m11) - (source.m10 * source.m01)) * s);
255     }
