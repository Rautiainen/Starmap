31      private void updateTextFields() {
32          jTextFieldCenterX.setText(String.format("%.3f", starSelection.getStarcube().getCenter().x));
33          jTextFieldCenterY.setText(String.format("%.3f", starSelection.getStarcube().getCenter().y));
34          jTextFieldCenterZ.setText(String.format("%.3f", starSelection.getStarcube().getCenter().z));
35          Vector3f rthetaphi = new Vector3f();
36          rthetaphi = starSelection.getStarcube().getCenterSpherical();
37          rthetaphi.siUnitstoAstronomical();
38          jTextFieldCenterDistance.setText(String.format("%.3f", rthetaphi.x));
39          jTextFieldCenterDec.setText(String.format("%.3f", rthetaphi.y));
40          jTextFieldCenterRAsc.setText(String.format("%.3f", rthetaphi.z));
41          jTextFieldObservationSpotX.setText(String.format("%.3f", starSelection.getStarcube().getObservationSpot().x));
42          jTextFieldObservationSpotY.setText(String.format("%.3f", starSelection.getStarcube().getObservationSpot().y));
43          jTextFieldObservationSpotZ.setText(String.format("%.3f", starSelection.getStarcube().getObservationSpot().z));
44          rthetaphi = starSelection.getStarcube().getObservationSpotSpherical();
45          rthetaphi.siUnitstoAstronomical();
46          jTextFieldObservationSpotDistance.setText(String.format("%.3f", rthetaphi.x));
47          jTextFieldObservationSpotDec.setText(String.format("%.3f", rthetaphi.y));
48          jTextFieldObservationSpotRAsc.setText(String.format("%.3f", rthetaphi.z));
49          jTextFieldEdgeLength.setText(String.format("%.3f", starSelection.getStarcube().getEdgelength()));
50      }

352         center = starSelection.getStarcube().getCenter();
353         float floatinput = center.x;
354         try {
355             floatinput = Float.parseFloat(jTextFieldCenterX.getText());

357             System.out.println("Error reading input: " + e.getMessage());
358         }
359         center = new Vector3f(floatinput, center.y, center.z);
360         starSelection.getStarcube().setCenter(center);

362     }//GEN-LAST:event_jTextFieldCenterXActionPerformed
